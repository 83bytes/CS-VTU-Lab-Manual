## Program: wordlength.c

	#include<stdio.h>
	#include<stdlib.h>

	int main(void)
	{

		int iLen=0;
		unsigned int i=~0;

		for(i = 1; i != 0; i = i << 1)
		{

			iLen++;
		}

		printf("\n\nThe Word length of the host machine is %d\n\n", iLen);
		return 0;
	}
